41      """ 
42      Example how to compare two runs and plot a histogram of the similarity of all the matching spectra. It can compare 
43      any two mzML files but works best when comparing two aligned files. It compares the retention time of all the spectra 
44      of both msruns with each other and when they are the same, uses pymzml's spec1.similarityTo(spec2) function. This returns 
45      a value between 0-1 for each compared spectrum which can then be plotted.  
46      Also possible is to put in 3 or more msrun files, this will compare all msruns with eachother (so: 1 to 2, 1 to 3 and 2 to 3.)  
47       
48       
49      This example script uses the following classes and functions: 
50        - L{compareRuns.CompareRuns} 
51        - L{compareRuns.CompareRuns.compare_mzML} 
52   
53       
54      Comparing two msruns and plotting a histogram of the similarity between the spectra 
55       
56      >>> from pyMSA import compareRuns                                     
57      >>> import pymzml 
58      >>> msrun1 = pymzml.run.Reader(r'example_files/input/example_aligned_file_1.aligned.mzML')                                    # Make an instance of pymzml.run.Reader with file you want to compare as input 
59      >>> msrun2 = pymzml.run.Reader(r'example_files/input/example_aligned_file_2.aligned.mzML')                                    # Make another instance of pymzml.run.Reader with file you want to compare as input 
60      >>> compare = compareRuns.CompareRuns(msrun1, msrun2)                                                                         # Make an instance of compareRuns.CompareRuns with the two instances of pymzml.run.Reader that you want to compare 
61      >>> similarityList, missCount1, missCount2 = compare.compare_mzML()                                                           # Compare the two files and retrieve a list of similarities 
62      >>> floatVector = R.FloatVector(similarityList)                                                                               # Make a FloatVector out of the list 
63      >>> plots = rPlots.Plots()                                                                                                    # Make an instance of rPlots.Plots 
64      >>> plots.histogram('example_files/output/similarity_histogram.png', floatVector,                                             # Plot a histogram of the results 
65      ...     xlab='similarity', ylab='# of spectra', width=600, height=600,  
66      ...     title='compare example 1 and example 2. Missed in first: '+str(missCount1)+' missed in second: '+str(missCount2)) 
67      """

71      # Make another instance of pymzml.run.Reader with file you want to compare as input 
72      msrun2 = pymzml.run.Reader(r'example_files/input/example_aligned_file_2.aligned.mzML')

74      # Make an instance of compareRuns.CompareRuns with the two instances of pymzml.run.Reader that you want to compare

78      # Make a FloatVector out of the list

80      # Make an instance of rPlots.Plots 
81      plots = rPlots.Plots()
